001package shoppingListGUI;
002
003import javax.swing.JMenuBar;
004import javax.swing.JMenu;
005import javax.swing.JMenuItem;
006import javax.swing.KeyStroke;
007import javax.swing.ImageIcon;
008import java.awt.event.ActionListener;
009import java.awt.event.KeyEvent;
010import java.awt.event.ActionEvent;
011import java.awt.event.WindowEvent;
012
013/**
014 * @author  <a href="mailto:vaino.niemi@cs.tamk.fi">Vaino Niemi</a>
015 * @version 2015.1204
016 * @since   1.8
017 */
018public class MyMenuBar extends JMenuBar {
019
020    /**
021     * References the main gui object.
022     */
023    private GUInterface gui;
024
025    /**
026     * Hold the new list button.
027     */
028    private JMenuItem newList;
029
030    /**
031     * Hold the save list button.
032     */
033    private JMenuItem saveList;
034
035    /**
036     * Hold the save to dropbox button.
037     */
038    private JMenuItem dropboxSave;
039
040    /**
041     * Hold the load from dropbox button.
042     */
043    private JMenuItem dropboxLoad;
044
045    /**
046     * Hold the load button.
047     */
048    private JMenuItem loadList;
049
050    /**
051     * Hold the exit button.
052     */
053    private JMenuItem exitButton;
054
055    /**
056     * Buils a new menu bar and the buttons in it.
057     *
058     * With the menu items you can control the GUI.
059     *
060     * @param gui the main object of the gui.
061     */
062    public MyMenuBar(GUInterface gui) {
063
064        super();
065
066        this.gui = gui;
067
068        createFileMenu();
069        createListeners();
070    }
071
072    /**
073     * Creates the file menu and the buttons in it.
074     *
075     * Few of the buttons have neumonics and shortcuts user can use.
076     */
077    public void createFileMenu() {
078
079        // Create file menu
080        JMenu file = new JMenu("File");
081        file.setMnemonic(KeyEvent.VK_F);
082        file.getAccessibleContext().setAccessibleDescription(
083        "Menu for loading or saving a shop list or quitting the app.");
084        add(file);
085
086        // ------
087        // Add items to the menus
088        // ------
089
090        // Get working dir -> res -> icons
091        String path = System.getProperties().getProperty("user.dir");
092        String sep = System.getProperties().getProperty("file.separator");
093        String icons = path + sep + "res" + sep + "icons" + sep;
094
095        // New - button
096        newList = new JMenuItem("New", KeyEvent.VK_N);
097        newList.setIcon(new ImageIcon(icons + "New16.gif"));
098        newList.setAccelerator(
099            KeyStroke.getKeyStroke(KeyEvent.VK_N, ActionEvent.CTRL_MASK));
100        newList.getAccessibleContext().setAccessibleDescription(
101        "Starts a blank shopping list.");
102        file.add(newList);
103
104        file.addSeparator();
105
106        // Save to PC - button
107        saveList = new JMenuItem("Save to PC...", KeyEvent.VK_S);
108        saveList.setIcon(new ImageIcon(icons + "SaveAs16.gif"));
109        saveList.setAccelerator(
110            KeyStroke.getKeyStroke(KeyEvent.VK_S, ActionEvent.CTRL_MASK));
111        saveList.getAccessibleContext().setAccessibleDescription(
112        "Saves the current list to a location on PC...");
113        file.add(saveList);
114
115        // Load - button
116        loadList = new JMenuItem("Open list from PC...",
117            KeyEvent.VK_O);
118        loadList.setIcon(new ImageIcon(icons + "Open16.gif"));
119        loadList.setAccelerator(
120            KeyStroke.getKeyStroke(KeyEvent.VK_O, ActionEvent.CTRL_MASK));
121        loadList.getAccessibleContext().setAccessibleDescription(
122        "Loads a list from a location...");
123        file.add(loadList);
124
125        file.addSeparator();
126
127        // Save to Dropbox - button
128        dropboxSave = new JMenuItem("Save to Dropbox...");
129        dropboxSave.setIcon(new ImageIcon(icons + "SaveAs16.gif"));
130        dropboxSave.getAccessibleContext().setAccessibleDescription(
131        "Saves the current list to dropbox...");
132        file.add(dropboxSave);
133
134        // Load from Dropbox - button
135        dropboxLoad = new JMenuItem("Open list from Dropbox...");
136        dropboxLoad.setIcon(new ImageIcon(icons + "Open16.gif"));
137        dropboxLoad.getAccessibleContext().setAccessibleDescription(
138        "Loads a list from a location...");
139        file.add(dropboxLoad);
140
141        file.addSeparator();
142
143        // Exit - button
144        exitButton = new JMenuItem("Close", KeyEvent.VK_X);
145        exitButton.setIcon(new ImageIcon(icons + "Stop16.gif"));
146        exitButton.setAccelerator(
147            KeyStroke.getKeyStroke(KeyEvent.VK_F4, ActionEvent.ALT_MASK));
148        exitButton.getAccessibleContext().setAccessibleDescription(
149        "Exits the application.");
150        file.add(exitButton);
151    }
152
153    /**
154     * Creates listeners for the button presses.
155     */
156    public void createListeners() {
157
158        exitButton.addActionListener(e -> gui.dispatchEvent(
159                    new WindowEvent(gui, WindowEvent.WINDOW_CLOSING)));
160
161        saveList.addActionListener(e -> gui.saveFile(false));
162
163        newList.addActionListener(e -> gui.saveFile(true));
164
165        loadList.addActionListener(e -> gui.loadFile());
166
167        dropboxLoad.addActionListener(e -> new DropboxLoadDialog(
168            gui));
169
170        dropboxSave.addActionListener(e -> new DropboxSaveDialog(
171            gui));
172    }
173}
