001package listControl;
002
003/**
004 * @author  <a href="mailto:vaino.niemi@cs.tamk.fi">Vaino Niemi</a>
005 * @version 2015.1127
006 * @since   1.8
007 */
008public interface MyList<T> {
009
010    /**
011     * Appends the specified element to the end of this list.
012     *
013     * @param e the element to be appended to the list
014     */
015    void add(T e);
016
017    /**
018     * Removes all of the elements from this list.
019     */
020    void clear();
021
022    /**
023     * Returns the element at the specified position in this list.
024     *
025     * @param index index of the element to return
026     * @return the object at the specified position in this list
027     */
028    T get(int index);
029
030    /**
031     * Returns true if this list contains no elements.
032     *
033     * @return true, if the list contains no elements.
034     */
035    boolean isEmpty();
036
037    /**
038     * Removes the element at the specified position in this list. 
039     *
040     * When the element that is going to be removed is found,
041     * the element is returned and the node holding the element
042     * will be left out of the list.
043     *
044     * @param index the index of the element to be removed
045     * @return the element previously at the specified location
046     */
047    T remove(int index);
048
049    /**
050     * Removes the first occurrence of the specified element from this list.
051     *
052     * Works only if the element is present (return true).
053     *
054     * @param o the element to be deleted from this list, if present
055     * @return true, if the list contained the specified element.
056     */
057    boolean remove(T o);
058
059    /**
060     * Returns the index of the first occurrence of the specified element.
061     * 
062     * Returns -1 if this list does not contain the element.
063     *
064     * @param o the element to look from the list.
065     * @return the index of the specified element or -1 if element doesn't
066     * exist.
067     */
068    int indexOf(T o);
069
070    /**
071     * Returns the number of elements in this list.
072     *
073     * @return the number of elements in this list.
074     */
075    int size();
076}
